```
class ResourcesSignature(dspy.Signature):
    """
    Analise o conjunto de conversas de WhatsApp de uma clínica odontológica e identifique:
    1. Os profissionais mencionados (dentistas, doutores, especialistas) com seus títulos.
    2. O tipo de agendamento praticado pela clínica.

    Responda em português. Baseie-se exclusivamente nos textos fornecidos.
    """
    conversations_sample: str = dspy.InputField(
        desc="Amostra de conversas da clínica (até 10 conversas, início e fim de cada uma)"
    )
    clinic_name: str = dspy.InputField(desc="Nome da clínica")
    professionals: list = dspy.OutputField(
        desc=(
            "Lista de nomes de profissionais detectados nas conversas. "
            "Inclua título quando presente (ex: ['Dra. Ana', 'Dr. Carlos']). "
            "Retorne lista vazia [] se nenhum profissional for identificado. "
            "NÃO invente nomes — inclua apenas os que aparecem no texto."
        )
    )
    schedule_type: str = dspy.OutputField(
        desc=(
            "Tipo de agendamento: "
            "'by_professional' se a clínica agenda por profissional específico; "
            "'by_room' se agenda por sala ou consultório; "
            "'single' se há apenas um profissional ou agendamento não menciona profissionais. "
            "Retorne exatamente um dos três valores: single, by_professional, by_room."
        )
    )
```

persist_resources(db, clinic_id, job_id, professionals, schedule_type, services):
  1. Delete unconfirmed resources: db.table("la_resources").delete().eq("clinic_id", clinic_id).eq("confirmed", False).execute()
  2. Delete unconfirmed services: db.table("la_services").delete().eq("clinic_id", clinic_id).eq("confirmed", False).execute()
  3. Build professional rows list; if empty, create one schedule_config row:
     {"clinic_id": clinic_id, "job_id": job_id, "resource_type": "schedule_config", "name": "default", "schedule_type": schedule_type, "confirmed": False}
  4. Batch insert professionals (if rows): db.table("la_resources").insert(rows).execute()
  5. Batch insert services (if services): db.table("la_services").insert(svc_rows).execute()
     Each svc row: {"clinic_id": clinic_id, "job_id": job_id, "name": svc["name"], "mention_count": svc["mention_count"], "confirmed": False}
